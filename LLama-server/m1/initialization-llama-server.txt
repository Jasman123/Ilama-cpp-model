# The most basic start command
# --model        : path to your GGUF file
# --port         : which port to listen on (8080 is convention)
# --n-gpu-layers : how many layers to put on the GPU (-1 = all layers = full GPU)

llama-server.exe \
  --model ./models/your-model.gguf \
  --port 8080 \
  --n-gpu-layers -1
